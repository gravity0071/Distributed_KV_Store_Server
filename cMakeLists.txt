cmake_minimum_required(VERSION 3.10)

project(Distributed_KV_Store_Master VERSION 1.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the executable before linking libraries to it
add_executable(MyExecutable main.cpp)

# FetchContent block for nlohmann_json
include(FetchContent)
FetchContent_Declare(
        nlohmann_json
        GIT_REPOSITORY https://github.com/nlohmann/json.git
        GIT_TAG v3.10.5  # Replace with the desired version
)
FetchContent_MakeAvailable(nlohmann_json)

# Link nlohmann_json to the executable
target_link_libraries(MyExecutable PRIVATE nlohmann_json::nlohmann_json)
